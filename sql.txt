CREATE TABLE Admin (
    managerid CHAR PRIMARY KEY NOT NULL,
    managerpw CHAR NULL,
    name CHAR NULL,
    count INT NULL
);

CREATE TABLE `User` (
	`userid`	Char	PRIMARY KEY NOT NULL,
	`gender`	Char	NULL,
	`name`	Char	NULL,
	`userpw`	Char	NULL,
	`address`	Char	NULL,
	`phone`	Char	NULL,
	`email`	Char	NULL
);

CREATE TABLE `Trainer` (
	`TrainerID`	Char	PRIMARY KEY NOT NULL,
	`gender`	Char	NULL,
	`name`	Char	NULL,
	`userpw`	Char	NULL,
	`address`	Char	NULL,
	`phone`	Char	NULL,
	`email`	Char	NULL,
	`certification`	Boolean	NULL
);

CREATE TABLE `Membership_PT` (
	`number`	Int	NULL,
	`locker`	Boolean	NULL,
	`uniform`	Boolean	NULL,
	`strarttime`	datetime	NULL,
	`lasttime`	datetime	NULL,
	FOREIGN KEY (userid) REFERENCES User(userid)
);

CREATE TABLE `Membership_normal` (
	`locker`	Boolean	NULL,
	`uniform`	Boolean	NULL,
	`strarttime`	Date	NULL,
	`lasttime`	Date	NULL,
	FOREIGN KEY (userid) REFERENCES User(userid)
);

CREATE TABLE `FreeBoard_rep` (
	`CommentID`	Char	PRIMARY KEY NOT NULL,
	`Content`	Char	NULL,
	`CreationDate`	datetime	NULL,
	FOREIGN KEY (PostID) REFERENCES FreeBoard(PostID),
	FOREIGN KEY (userid) REFERENCES User(userid),
	FOREIGN KEY (TrainerID) REFERENCES Trainer(TrainerID)
);

CREATE TABLE `MatchingBoard_rep` (
	`CommentID`	Char	PRIMARY KEY NOT NULL,
	`PostID`	Char	NOT NULL,
	`userid`	Char	NOT NULL,
	`Content`	Char	NULL,
	`CreationDate`	datetime	NULL,
	FOREIGN KEY (userid) REFERENCES User(userid),
	FOREIGN KEY (PostID) REFERENCES MatchingBoard(PostID)
);

CREATE TABLE `Notice` (
	`PostID`	Char	PRIMARY KEY NOT NULL,
	`Title`	Char	NULL,
	`Content`	Char	NULL,
	`CreationDate`	Char	NULL,
	FOREIGN KEY (adminid) REFERENCES admin(adminid)

);

CREATE TABLE `FreeBoard` (
	`PostID`	Char	PRIMARY KEY NOT NULL,
	`Title`	VarChar	NULL,
	`Content`	VarChar	NULL,
	`CreationDate`	time	NULL,
	`repCount`	int	NULL,
	`likeCount`	int	NULL,
	FOREIGN KEY (userid) REFERENCES User(userid),
	FOREIGN KEY (TrainerID) REFERENCES Trainer(TrainerID)

);

CREATE TABLE `OT` (
	`ReservationID`	Char	PRIMARY KEY NOT NULL,
	`ReservationDate`	datetime	NOT NULL,
	`ReservationTime`	datetime	NULL,
	FOREIGN KEY (userid) REFERENCES User(userid),
	FOREIGN KEY (TrainerID) REFERENCES Trainer(TrainerID)
);

CREATE TABLE `OnetoOne` (
	`ReservationID`	Char	PRIMARY KEY NOT NULL,
	`ReservationDate`	datetime	NULL,
	`ReservationTime`	datetime	NULL,
	FOREIGN KEY (userid) REFERENCES User(userid),
	FOREIGN KEY (TrainerID) REFERENCES Trainer(TrainerID)
);

CREATE TABLE `PTAttendance` (
	`Count`	INT	AUTO_INCREMENT PRIMARY KEY NOT NULL,
	`Check_in_Day`	Char	NULL,
	FOREIGN KEY (userid) REFERENCES User(userid),
	FOREIGN KEY (TrainerID) REFERENCES Trainer(TrainerID)
);

CREATE TABLE `Schedule` (
	`Holiday`	Date	NULL,
	FOREIGN KEY (TrainerID) REFERENCES Trainer(TrainerID)
);

CREATE TABLE `NomalAttendance` (
	`Check_in_Day`	Char	NULL,
	FOREIGN KEY (userid) REFERENCES User(userid)
);

CREATE TABLE `MatchingBoard` (
	`PostID`	Char	PRIMARY KEY NOT NULL,
	`Title`		VarChar	NULL,
	`Content`	VarChar	NULL,
	`CreationDate`	time	NULL,
	FOREIGN KEY (userid) REFERENCES User(userid)
);

CREATE TABLE `GroupPT` (
	`ReservationID`	Char	PRIMARY KEY NOT NULL,
	`personnel`	Char	NULL,
	`Reservation time`	Char	NULL,
	FOREIGN KEY (userid) REFERENCES User(userid)
);
